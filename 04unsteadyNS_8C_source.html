80int main(int argc, char* argv[])
81{
82 // Construct the tutorial04 object
83 tutorial04 example(argc, argv);
84 // Read parameters from ITHACAdict file
85 ITHACAparameters* para = ITHACAparameters::getInstance(example._mesh(),
86 example._runTime());
87 int NmodesUout = para->ITHACAdict->lookupOrDefault<int>("NmodesUout", 15);
88 int NmodesPout = para->ITHACAdict->lookupOrDefault<int>("NmodesPout", 15);
89 int NmodesSUPout = para->ITHACAdict->lookupOrDefault<int>("NmodesSUPout", 15);
90 int NmodesUproj = para->ITHACAdict->lookupOrDefault<int>("NmodesUproj", 10);
91 int NmodesPproj = para->ITHACAdict->lookupOrDefault<int>("NmodesPproj", 10);
92 int NmodesSUPproj = para->ITHACAdict->lookupOrDefault<int>("NmodesSUPproj", 10);
94 example.Pnumber = 1;
96 example.Tnumber = 1;
98 example.setParameters();
99 // Set the parameter ranges
100 example.mu_range(0, 0) = 0.005;
101 example.mu_range(0, 1) = 0.005;
102 // Generate equispaced samples inside the parameter range
103 example.genEquiPar();
104 // Set the inlet boundaries where we have non homogeneous boundary conditions
105 example.inletIndex.resize(1, 2);
106 example.inletIndex(0, 0) = 0;
107 example.inletIndex(0, 1) = 0;
108 // Time parameters
109 example.startTime = 50;
110 example.finalTime = 70;
111 example.timeStep = 0.01;
112 example.writeEvery = 0.1;
113 // Perform The Offline Solve;
114 example.offlineSolve();
115 // Solve the supremizer problem
116 example.solvesupremizer();
117 // Search the lift function
118 example.liftSolve();
119 // Normalize the lifting function
120 ITHACAutilities::normalizeFields(example.liftfield);
121 // Create homogeneous basis functions for velocity
122 example.computeLift(example.Ufield, example.liftfield, example.Uomfield);
123 // Perform a POD decomposition for velocity and pressure
124 ITHACAPOD::getModes(example.Uomfield, example.Umodes, example._U().name(),
125 example.podex, 0, 0,
126 NmodesUout);
127 ITHACAPOD::getModes(example.Pfield, example.Pmodes, example._p().name(),
128 example.podex, 0, 0,
129 NmodesPout);
130 ITHACAPOD::getModes(example.supfield, example.supmodes, example._U().name(),
131 example.podex,
132 example.supex, 1, NmodesSUPout);
133 example.projectSUP("./Matrices", NmodesUproj, NmodesPproj, NmodesSUPproj);
134 reducedUnsteadyNS reduced(example);
135 // Set values of the reduced stuff
136 reduced.nu = 0.005;
137 reduced.tstart = 50;
138 reduced.finalTime = 70;
139 reduced.dt = 0.005;
140 reduced.storeEvery = 0.005;
141 reduced.exportEvery = 0.1;
142 // Set the online velocity
143 Eigen::MatrixXd vel_now(1, 1);
144 vel_now(0, 0) = 1;
145 reduced.solveOnline_sup(vel_now, 1);
146 // Reconstruct the solution and export it
147 reduced.reconstruct(true, "./ITHACAoutput/ReconstructionSUP/");
148 exit(0);
149}
